#### Native Ansible
Kali:
ansible-playbook deploy.yml -i "192.168.1.100," -e "ansible_user=kali ansible_connection=ssh machine=kali"

Windows:
ansible-playbook deploy.yml -i "192.168.6.153," -e "ansible_user=ansible ansible_connection=winrm ansible_winrm_transport=ntlm ansible_port=5985 machine=windows"

### Docker Ansible
Kali:
docker run --rm -it -v "$(pwd)":/ansible -w /ansible willhallonline/ansible:latest \
  sh -c 'ansible-playbook deploy.yml -i "192.168.1.100," -e "ansible_user=kali ansible_connection=ssh machine=kali"'


Windows:
docker run --rm -it -v "$(pwd)":/ansible -w /ansible willhallonline/ansible:latest \
  sh -c 'ansible-playbook deploy.yml -i "192.168.6.153," -e "ansible_user=ansible ansible_connection=winrm ansible_winrm_transport=ntlm ansible_port=5985 machine=windows"'
